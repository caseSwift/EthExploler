import InputFields from './InputFields';
import { Link, Route } from 'react-router-dom';
import DatePicker from './DatePicker';



function App() {
    const [showDatePicker, setShowDatePicker] = useState(false);
    const [showInputFields, setShowInputFields] = useState(true);
    /** on submit get address value and starting block */
    const handleSubmit = (walletAddress, startingBlock) => {
        // This function will handle the submit event with the wallet address and starting block
        console.log(walletAddress, startingBlock);
    };
    /** handling visibility of diffrent elements */
    const handleButtonClick = () => {
        setShowDatePicker(true);
        setShowInputFields(false);
    };
    return (
        <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', justifyContent: 'center', height: '100vh' }}>
            {showInputFields && <InputFields onSubmit={handleSubmit} />}
            <div style={{ marginTop: '20px' }}>
                {showInputFields &&<Button variant="contained" onClick={handleButtonClick}>Click for more options</Button>}
                {showDatePicker && <DatePicker onClose={() => setShowDatePicker(false)} />}
            </div>
        </div>
    );
}

export default App;
