import React from 'react';
import { TextField, Button,MenuItem,Select } from '@mui/material';
import  { useState } from 'react';


function DatePicker() {
    const [coin, setCoin] = useState('eth');
    const [balance, setBalance] = useState(0); // initialize balance to 0
/** making request and doing validation to balance adress */
    const handleSubmit = async (date, address) => {
        const isValidAddress = /^0x[a-fA-F0-9]{40}$/.test(address);

        if (!isValidAddress) {
            alert("Please enter a valid Ethereum address.");
            return;
        }
        try {
            const response = await fetch(`http://localhost:3000/balance/${address}/${date}`);
            const data = await response.json();
            console.log(data); // log the response data to console for debugging

            setBalance(data.balance); // set the balance received from the server in state
        } catch (error) {
            console.error(error);
        }
    };

    return (
        <div>
            <h1>Select Date, Coin, and Address</h1>
            <form onSubmit={(e) => { e.preventDefault(); handleSubmit(e.target.date.value, e.target.address.value) }}>
                <TextField id="date" label="Date" type="date" InputLabelProps={{ shrink: true }} style={{ marginRight: '20px' }} inputProps={{ max: new Date().toISOString().split('T')[0] }} />
                <Select labelId="coin-select-label" id="coin-select" value={coin} onChange={(e) => setCoin(e.target.value)} style={{ marginRight: '20px' }}>
                    <MenuItem value="eth">Ethereum</MenuItem>

                </Select>
                <TextField id="address" label="Address" variant="outlined" style={{ marginRight: '20px' }} />
                <Button variant="contained" type="submit">Submit</Button>
            </form>
            {balance > 0 && <p>Balance: {balance}</p>} {/* display balance if it's greater than 0 */}
        </div>
    );
}


export default DatePicker;
